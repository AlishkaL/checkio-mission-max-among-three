For three numbers it's just checking two conditions. 

function maxOfThree(a: number, b: number, c: number): number {

    if (a >= b && a >= c) return a;
    if (b >= a && b >= c) return b;
    return c;
}

Using reduce method which makes the same checking and keep the biggest element on every step. 

function maxOfThree(a: number, b: number, c: number): number {

    return [a, b, c].reduce((max, curr) => (curr > max ? curr : max), Number.MIN_SAFE_INTEGER);
}
